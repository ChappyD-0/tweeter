// Get elements
const spellCard = document.querySelector(".spell-card");
const cardSvg = document.querySelector(".card-image svg");
const body = document.body;
const particles = document.querySelectorAll(".particle");

// --- Particle Animation ---

// Get card position and size to determine orbit center
let cardRect = spellCard.getBoundingClientRect();
let centerX = cardRect.left + cardRect.width / 2;
let centerY = cardRect.top + cardRect.height / 2;

// Function to get a random number within a range
function getRandom(min, max) {
	return Math.random() * (max - min) + min;
}

// Store parameters for each particle
const particleParams = [];
particles.forEach(() => {
	particleParams.push({
		radiusX: getRandom(150, 300), // Random horizontal radius
		radiusY: getRandom(150, 300), // Random vertical radius
		speed: getRandom(0.001, 0.005), // Random speed (radians per frame)
		angleOffset: getRandom(0, Math.PI * 2), // Random starting angle
		currentAngle: getRandom(0, Math.PI * 2) // Initial position angle
	});
});

// Animation loop
function animateParticles() {
	// Recalculate card center in case of resize or layout change (though card is fixed)
	cardRect = spellCard.getBoundingClientRect();
	centerX = cardRect.left + cardRect.width / 2;
	centerY = cardRect.top + cardRect.height / 2;

	particles.forEach((particle, index) => {
		const params = particleParams[index];

		// Update the angle
		params.currentAngle += params.speed;

		// Calculate the new position on the ellipse
		// x = centerX + radiusX * cos(angle)
		// y = centerY + radiusY * sin(angle)
		const x =
			centerX +
			params.radiusX * Math.cos(params.currentAngle + params.angleOffset);
		const y =
			centerY +
			params.radiusY * Math.sin(params.currentAngle + params.angleOffset);

		// Apply the position to the particle
		// Using left/top relative to body, combined with transform: translate(-50%, -50%) in CSS
		particle.style.left = `${x}px`;
		particle.style.top = `${y}px`;

		// Optional: Add a slight scale animation for twinkling effect
		// particle.style.transform = `translate(-50%, -50%) scale(${Math.sin(params.currentAngle * 5) * 0.1 + 0.9})`;
	});

	requestAnimationFrame(animateParticles);
}

// Start the animation
animateParticles();

// --- Card Click Interaction ---

// Add a class on click
spellCard.addEventListener("click", function () {
	console.log("Card clicked!");

	// Add flash effect
	body.classList.add("flash");
	// Remove flash effect after a short delay
	setTimeout(() => {
		body.classList.remove("flash");
	}, 300); // Match the transition duration in CSS

	// Add explore effect to SVG
	cardSvg.classList.add("explore");
	// Remove explore effect after a short delay
	setTimeout(() => {
		cardSvg.classList.remove("explore");
	}, 500); // Duration for SVG effect
});
